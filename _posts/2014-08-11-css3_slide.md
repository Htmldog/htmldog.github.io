<script type="text/javascript">
var a=box.getElementsByTagName('a');
var num=-1;
function nextCard(m){
    for(var i=0;i<4;i++){
        a[i].className='out';
    }
    a[m].className='in';
}
bt.onclick=box.onclick=function(){
    num++;
    if(num>=4){
        num=0;
    }
    nextCard(num);
    return false;
}
box.click();
</script>
